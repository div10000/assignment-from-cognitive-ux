copy((() => {
  const sec = document.querySelector('section.exclusive');
  const secCS = getComputedStyle(sec);
  const prev = document.querySelector('.swiper-button-prev');
  const next = document.querySelector('.swiper-button-next');
  const pag = document.querySelector('.swiper-pagination');
  const bullet = document.querySelector('.swiper-pagination-bullet');
  const slides = [...document.querySelectorAll('.swiper-wrapper > .swiper-slide .slide')];

  return JSON.stringify({
    sectionBackground: secCS.backgroundImage,
    sectionBackgroundColor: secCS.backgroundColor,
    sectionPadding: secCS.padding,
    arrowPrevCSS: prev ? getComputedStyle(prev).cssText.slice(0, 0) || {
      width: getComputedStyle(prev).width, height: getComputedStyle(prev).height,
      background: getComputedStyle(prev).backgroundColor, color: getComputedStyle(prev).color,
      position: getComputedStyle(prev).position, left: getComputedStyle(prev).left
    } : null,
    arrowNextLeft: next ? getComputedStyle(next).right : null,
    paginationCSS: pag ? { bottom: getComputedStyle(pag).bottom } : null,
    bulletCSS: bullet ? {
      width: getComputedStyle(bullet).width, height: getComputedStyle(bullet).height,
      background: getComputedStyle(bullet).backgroundColor, opacity: getComputedStyle(bullet).opacity
    } : null,
    slideFullHTML: slides.map(s => s.outerHTML)
  }, null, 2);
})())
