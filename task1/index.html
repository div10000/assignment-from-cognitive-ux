Where this stands now vs. what's still needed

Now exact (from your two console dumps):

  • All 4 cards' real text + real logo images (ixigo, Reliance Digital, BookMyShow) and the "*available on selected cards" disclaimer on the Enjoy Benefits card.
  • Arrow buttons: 30×30px, transparent background, positioned 10px from the left/right edge.
  • Pagination: sits 8px from the bottom; bullets are 10×10px, color #e3530f.

One genuinely surprising result: section.exclusive itself has background: none and a transparent background-color. So the peach/tan textured background you see is not on this element — it has to be coming from a parent (maybe <body>), or from a ::before/::after pseudo-element on the section, which a plain getComputedStyle(section) call can't see. The snippet below checks both places.

Also still unconfirmed: the arrow icon's actual rendering (computed color came back as Swiper's literal default blue, #007aff — possibly never themed, possibly rendered via a pseudo-element my snippet didn't check), in-card type sizes, and the bullet's inactive state color (I only got one bullet's values, not the active vs. inactive distinction).

Paste this into the live page's console and send me the clipboard JSON:

copy((() => {
  const sec = document.querySelector('section.exclusive');
  const before = getComputedStyle(sec, '::before');
  const after = getComputedStyle(sec, '::after');
  const body = getComputedStyle(document.body);
  const html = getComputedStyle(document.documentElement);
  const prev = document.querySelector('.swiper-button-prev');
  const prevAfter = prev ? getComputedStyle(prev, '::after') : null;
  const bullets = [...document.querySelectorAll('.swiper-pagination-bullet')];
  const activeBullet = document.querySelector('.swiper-pagination-bullet-active');

  return JSON.stringify({
    bodyBackground: { image: body.backgroundImage, color: body.backgroundColor },
    htmlBackground: { image: html.backgroundImage, color: html.backgroundColor },
    sectionBeforePseudo: { content: before.content, image: before.backgroundImage, color: before.backgroundColor },
    sectionAfterPseudo: { content: after.content, image: after.backgroundImage, color: after.backgroundColor },
    arrowPrevBackgroundImage: prev ? getComputedStyle(prev).backgroundImage : null,
    arrowPrevAfterPseudo: prevAfter ? { content: prevAfter.content, fontFamily: prevAfter.fontFamily, color: prevAfter.color } : null,
    bulletDefault: bullets.length ? { background: getComputedStyle(bullets[0]).backgroundColor, opacity: getComputedStyle(bullets[0]).opacity } : null,
    bulletActive: activeBullet ? { background: getComputedStyle(activeBullet).backgroundColor, opacity: getComputedStyle(activeBullet).opacity, transform: getComputedStyle(activeBullet).transform } : null,
    cardTextSizes: (() => {
      const h3 = document.querySelector('.slide h3');
      const p = document.querySelector('.slide p');
      const em = document.querySelector('.slide em');
      return {
        h3: h3 ? { fontSize: getComputedStyle(h3).fontSize, fontWeight: getComputedStyle(h3).fontWeight, margin: getComputedStyle(h3).margin, color: getComputedStyle(h3).color } : null,
        p: p ? { fontSize: getComputedStyle(p).fontSize, lineHeight: getComputedStyle(p).lineHeight, margin: getComputedStyle(p).margin } : null,
        em: em ? { fontSize: getComputedStyle(em).fontSize, color: getComputedStyle(em).color } : null
      };
    })()
  }, null, 2);
})())
